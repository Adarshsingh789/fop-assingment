#include <stdio.h>

long fact(int n) {
    long f=1;
    for(int i=1;i<=n;i++) f*=i;
    return f;
}

int main() {
    int n, i;
    float x, sum = 0;

    printf("Enter x (radians): ");
    scanf("%f",&x);
    printf("Enter terms: ");
    scanf("%d",&n);

    for(i=0;i<n;i++) {
        float term = pow(x, 2*i+1)/fact(2*i+1);
        if(i%2==0)
            sum += term;
        else
            sum -= term;
    }

    printf("Sum = %f\n", sum);
    return 0;
}

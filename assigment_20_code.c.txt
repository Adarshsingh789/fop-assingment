#include <stdio.h>

void swap_val(int a, int b) {
    int temp = a;
    a = b;
    b = temp;
    printf("Without pointers: a=%d b=%d\n", a, b);
}

void swap_ref(int *x, int *y) {
    int temp = *x;
    *x = *y;
    *y = temp;
}

int main() {
    int a, b;

    printf("Enter numbers: ");
    scanf("%d %d",&a,&b);

    swap_val(a,b);

    swap_ref(&a,&b);
    printf("With pointers: a=%d b=%d\n", a, b);

    return 0;
}

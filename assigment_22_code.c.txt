#include <stdio.h>
#include <string.h>

int main() {
    char s1[100], s2[100];
    int choice;

    printf("Enter string1: ");
    scanf("%s", s1);
    printf("Enter string2: ");
    scanf("%s", s2);

    printf("1.Length\n2.Copy\n3.Concat\n4.Compare\n");
    scanf("%d",&choice);

    switch(choice) {
        case 1:
            printf("Length = %lu\n", strlen(s1));
            break;
        case 2:
            strcpy(s2, s1);
            printf("Copied: %s\n", s2);
            break;
        case 3:
            strcat(s1, s2);
            printf("Concatenated: %s\n", s1);
            break;
        case 4:
            if(strcmp(s1,s2)==0)
                printf("Equal\n");
            else
                printf("Not equal\n");
            break;
    }

    return 0;
}

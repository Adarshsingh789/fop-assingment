#include <stdio.h>
#include <math.h>

int isPrime(int n) {
    if(n < 2) return 0;
    for(int i = 2; i <= n/2; i++)
        if(n % i == 0) return 0;
    return 1;
}

int main() {
    int n, i, choice;

    printf("Enter number: ");
    scanf("%d", &n);

    printf("1.Square\n2.Cube\n3.Sqrt\n4.Prime\n5.Factorial\n6.Prime Factors\n");
    scanf("%d", &choice);

    switch(choice) {
        case 1: printf("Square = %d\n", n*n); break;
        case 2: printf("Cube = %d\n", n*n*n); break;
        case 3: printf("Square root = %.2f\n", sqrt(n)); break;

        case 4:
            if(isPrime(n))
                printf("Prime\n");
            else
                printf("Not Prime\n");
            break;

        case 5: {
            long fact = 1;
            for(i = 1; i <= n; i++)
                fact *= i;
            printf("Factorial = %ld\n", fact);
            break;
        }

        case 6:
            printf("Prime factors: ");
            for(i = 2; i <= n; i++) {
                while(n % i == 0) {
                    printf("%d ", i);
                    n /= i;
                }
            }
            break;

        default: printf("Invalid\n");
    }

    return 0;
}
